export function getDocumentClient(): DC {
  const dynamoDbConfig: ClientConfiguration = getDynamoDbConfig(
    process.env.REGION,
  )
 
  const DynamoDB = process.env.IS_OFFLINE
    ? AWSSdk.DynamoDB
    : AWSXRay.captureAWS(AWSSdk).DynamoDB
 
  return new DynamoDB.DocumentClient(dynamoDbConfig)
}

function generateExpressions({ expr, parentKey, value }) {
  if (isArray(value)) {
    return each(value, (val, index) =>
      generateExpressions({
        expr,
        parentKey: `${parentKey}[${index}]`,
        value: val,
      }),
    )
  }
 
  if (isObject(value)) {
    return each(value, (val, key) =>
      generateExpressions({
        expr,
        parentKey: `${parentKey}.${key}`,
        value: val,
      }),
    )
  }
 
  expr.set(parentKey, value)
}
