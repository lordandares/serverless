import { schemas } from '@lighthouse/serverless-common'
import { pick } from 'lodash'
import * as moment from 'moment'
import { v4 as uuid } from 'uuid'
 
const dataProperties = [
  'areas',

export function documentToSchedulePayload(
  document: schemas.ScheduleDocumentSchema,
): schemas.SchedulePayloadSchema {
  const { data, ...otherProps } = document
  const { scheduleId } = otherProps
 
  const payload: schemas.SchedulePayloadSchema = pick(
    {
      ...data,
      ...otherProps,
      id: scheduleId,
    },
    payloadProperties,
  )
 
  return payload
}
 
export function payloadToScheduleDocument(
  payload: schemas.SchedulePayloadSchema,
): schemas.ScheduleDocumentSchema {
  const {
    applicationId,
    createdAt,
    createdBy,
    endAt,
    id,
    startAt,
    userId,
  } = payload
 
  const isNew: boolean = !id
 
  const now: string = moment
    .utc()
    .toDate()
    .toISOString()
 
  // TODO: attach appropriate user label
  const actor = { id: userId, label: 'Unknown User', type: 'user' }
 
  const document: schemas.ScheduleDocumentSchema = {
    applicationId,
    createdAt: isNew ? now : createdAt,
    createdBy: isNew ? actor : createdBy,
    data: pick(payload, dataProperties),
    endAt,
    groupType: 'schedule',
    pk: `${applicationId}-schedule`,
    scheduleId: isNew ? uuid() : id,
    sk: `${isNew ? now : createdAt}-schedule`,
    startAt,
    updatedAt: now,
    updatedBy: actor,
  }
 
  return document
}
 
export function payloadToScheduleLocationDocument(payload: any): any {
  const { applicationId, locationId, name, serviceHours } = payload
 
  const now: string = moment
    .utc()
    .toDate()
    .toISOString()
 
  const document: any = {
    applicationId,
    createdAt: now,
    data: { name, serviceHours },
    groupType: 'location',
    pk: applicationId,
    sk: `location-${locationId}`,
    updatedAt: now,
  }
 
  return document
}
