import { errors, mongo, schemas } from '@lighthouse/serverless-common'
import { DynamoDB, SNS } from 'aws-sdk'
import * as BPromise from 'bluebird'
import { first, isEmpty, map, pick, toString } from 'lodash'
 
import { DynamoRepository } from '../repository/dynamo'
import {
  documentToSchedulePayload,
  payloadToScheduleDocument,
  payloadToScheduleLocationDocument,
} from '../service/lib/transform'

  env: {
    TABLE_SCHEDULES: string
    UPSERT_LOCATION_ARN: string
  }
}
 
interface CreateLocation {

  name: string
  serviceHours: object
}
 
const createLocation = async ({
  applicationId,
  locationId,
  name,
  serviceHours,
}: CreateLocation): Promise<schemas.LocationSchedulePayloadSchema> => {
  const table: string = process.env.TABLE_SCHEDULES

}
 
interface CreateSchedule {
  applicationId: string
  body: schemas.SchedulePayloadSchema
  userId: string
}
 
const createSchedule = async ({
  applicationId,
  body,
  userId,
}: CreateSchedule): Promise<schemas.SchedulePayloadSchema> => {
  const table: string = process.env.TABLE_SCHEDULES
  const dbObject = DynamoRepository(table)
 
  if (!applicationId) {
    throw new errors.ApplicationError({
      message: `The \`applicationId\` is missing from headers`,
    })
  }
 
  if (!userId) {
    throw new errors.ApplicationError({
      message: `The \`userId\` is missing from headers`,
    })
  }
 
  if (!body) {
    throw new errors.ValidationError({
      message: `A schedule body is required`,
    })
  }
 
  await schemas.validate({
    data: body,
    options: { context: { isNew: true } },
    schema: schemas.schedulePayloadSchema,
  })
 
  const areasCollection = await mongo.getCollection('areas')
  const { locations: scheduleLocations } = body
 
  const sns = new SNS()
  const topicArn = process.env.UPSERT_LOCATION_ARN
 
  const locations = await BPromise.reduce(
    scheduleLocations,
    async (accum, id) => {
      const area = await areasCollection.findOne({
        _id: new mongo.ObjectId(id),
      })
 
      if (!area) {
        return accum
      }
 
      const { application, name, serviceHours } = area
 
      sns
        .publish({
          Message: JSON.stringify({
            applicationId: toString(application),
            locationId: id,
            name,
            serviceHours,
          }),
          TopicArn: topicArn,
        })
        .promise()
 
      accum[id] = { name }
 
      return accum
    },
    {},
  )
 
  const included = { ...locations }
  const data = { ...body, applicationId, included, userId }
  const document: schemas.ScheduleDocumentSchema = payloadToScheduleDocument(
    data,
  )
 
  try {
    await schemas.validate({
      data: document,
      options: { context: { isNew: true } },
      schema: schemas.scheduleDocumentSchema,
    })
  } catch (err) {
    throw new errors.ApplicationError({ message: err.message })
  }
 
  const result: schemas.ScheduleDocumentSchema = await dbObject.create(document)
  const payload: schemas.SchedulePayloadSchema = documentToSchedulePayload(
    result,
  )
  return payload
}
 
interface GetSchedule {
  applicationId: string
  scheduleId: string | null
}
 
const getSchedule = async ({
  applicationId,
  scheduleId,
}: GetSchedule): Promise<schemas.SchedulePayloadSchema> => {
  const table: string = process.env.TABLE_SCHEDULES
  const dbObject = DynamoRepository(table)
 
  if (!applicationId) {
    throw new errors.ApplicationError({
      message: `The \`applicationId\` is missing from headers`,
    })
  }
 
  if (!scheduleId) {
    throw new errors.ValidationError({
      message: `The schedule \`id\` is required`,
    })
  }
 
  const result: schemas.ScheduleDocumentSchema[] = await dbObject.getSchedule(
    scheduleId,
  )
 
  if (isEmpty(result)) {
    throw new errors.ResourceNotFoundError({
      id: scheduleId,
      resource: 'Schedule',
    })
  }
 
  const document: schemas.ScheduleDocumentSchema = first(result)
  const payload: schemas.SchedulePayloadSchema = documentToSchedulePayload(
    document,
  )
  return payload
}

interface UpdateSchedule {
  applicationId: string
  body: schemas.SchedulePayloadSchema
  scheduleId: string | null
  userId: string
}
 
const updateSchedule = async ({
  applicationId,
  body,
  scheduleId,
  userId,
}: UpdateSchedule): Promise<schemas.SchedulePayloadSchema> => {
  const table: string = process.env.TABLE_SCHEDULES
  const dbObject = DynamoRepository(table)
 
  if (!applicationId) {
    throw new errors.ApplicationError({
      message: `The \`applicationId\` is missing from headers`,
    })
  }
 
  if (!userId) {
    throw new errors.ApplicationError({
      message: `The \`userId\` is missing from headers`,
    })
  }
 
  if (!scheduleId) {
    throw new errors.ValidationError({
      message: `The schedule \`id\` is required`,
    })
  }
 
  if (!body) {
    throw new errors.ValidationError({
      message: `A schedule body is required`,
    })
  }
 
  await schemas.validate({
    data: body,
    options: { context: { isNew: false } },
    schema: schemas.schedulePayloadSchema,
  })
 
  const result: schemas.ScheduleDocumentSchema[] = await dbObject.getSchedule(
    scheduleId,
  )
 
  if (isEmpty(result)) {
    throw new errors.ResourceNotFoundError({
      id: scheduleId,
      resource: 'Schedule',
    })
  }
 
  const existingDocument: schemas.ScheduleDocumentSchema = first(result)
  const existingDocumentPayload: schemas.SchedulePayloadSchema = documentToSchedulePayload(
    existingDocument,
  )
 
  const data = {
    // NOTE: spread existing document so we can transform below
    ...existingDocumentPayload,
    ...body,
    // NOTE: prevent user from overwriting original properties
    applicationId,
    createdAt: existingDocument.createdAt,
    createdBy: existingDocument.createdBy,
    id: scheduleId,
    userId,
  }
 
  const document: schemas.ScheduleDocumentSchema = payloadToScheduleDocument(
    data,
  )
 
  try {
    await schemas.validate({
      data: document,
      options: { context: { isNew: false } },
      schema: schemas.scheduleDocumentSchema,
    })
  } catch (err) {
    throw new errors.ApplicationError({ message: err.message })
  }
 
  const updatedDocument: schemas.ScheduleDocumentSchema = await dbObject.upsert(
    document,
  )
  const payload: schemas.SchedulePayloadSchema = documentToSchedulePayload(
    updatedDocument,
  )
  return payload
}
