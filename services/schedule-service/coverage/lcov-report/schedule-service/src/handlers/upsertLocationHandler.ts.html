import { errors, schemas } from '@lighthouse/serverless-common'
import { SNSEvent } from 'aws-lambda'
import { ScheduleService } from '../service/ScheduleService'
import parseSnsEvent from './lib/parseSnsEvent'
 
export async function upsertLocationHandler(event: SNSEvent) {
  try {
    const { body } = parseSnsEvent(event)
 
    const location: schemas.LocationSchedulePayloadSchema = await ScheduleService.createLocation(
      body,
    )
 
    return location
  } catch (err) {
    console.error('UpsertLocationError', {
      err,
      event,
    })
 
    if (errors.isKnownError(err)) {
      throw err
    }
 
    throw new errors.UnknownError()
  }
}
